 45    def __init__(self, path: str):
 46
 47        # Saving practical variables
 48        self.path = path                                                # The current output
 49        self.path_process = os.path.join(self.path, "processed")        # Raw processed data path
 50        self.path_files = os.path.join(self.path, "stats", "*.pt")      # The files query
 51        self.all_f = sorted(glob.iglob(self.path_files))                # The list of files
 52
 53        # Loading all files
 54        self.device = torch.device("cpu")                               # Ensuring the processing in on CPU
 55        self.loaded_f = [torch.load(f, map_location = self.device) for f in self.all_f]
 56
 57        self.thr_0 = cst.THRESHOLDS[0].item()                           # The zeroth entry in threshold
 58
 59        if os.path.exists(self.path_process):                           # Check if the dir exists
 60            shutil.rmtree(self.path_process)                            # We remove
 61        os.mkdir(self.path_process)                                     # We re-create it
 62
 63        self.stats = {}
 64
 65        for thr in cst.THRESHOLDS:
 66
 67            thr = thr.item()
 68
 69            # Extracting the data from the files
 70            for idx, loaded in enumerate(self.loaded_f):
 71                
 72                # If we are on the first iter, we create the basic tensor
 73                if idx == 0:
 74                    self.stats[thr] = {n: self._tensor_from_dict_val(n, loaded, thr) for n in METRICS_NAME}
 75
 76                # If not, we stack
 77                else:
 78                    for n, s in self.stats[thr].items():
 79                        self.stats[thr][n] = torch.vstack(( s, self._tensor_from_dict_val(n, loaded, thr) ))
 80        
 81        # Epoch counter
 82        self.epochs = torch.arange(0, len(self.stats[self.thr_0]["loss"])) + 1

328    def threshold_values(self):
329
330        computed = {
331            "iou": torch.stack([self.stats[thr.item()]["iou"].mean(dim = 1) for thr in cst.THRESHOLDS]),
332            "f1": torch.stack([self.stats[thr.item()]["f1"].mean(dim = (1, 2)) for thr in cst.THRESHOLDS]),
333            "ap": torch.stack([self.stats[thr.item()]["ap"].mean(dim = 1) for thr in cst.THRESHOLDS]),
334            "pre": torch.stack([self.stats[thr.item()]["precision"].mean(dim = (1, 2)) for thr in cst.THRESHOLDS]),
335            "rec": torch.stack([self.stats[thr.item()]["recall"].mean(dim = (1, 2)) for thr in cst.THRESHOLDS])
336        }
337        
338        for metric in computed.keys():
339
340            pl.figure()
341
342            data = computed[metric].T
343
344            for idx, epoch in enumerate(data):
345
346                if idx < len(data) - 1:
347                    pl.plot(cst.THRESHOLDS, epoch, color = "black", alpha = idx / len(data), linewidth = 0.6)
348                else:
349                    pl.plot(cst.THRESHOLDS, epoch, color = "crimson", alpha = idx / len(data), label = f"Last epoch")
350
351            pl.legend(loc = "lower center")
352            pl.xlabel("Threshold value")
353            pl.ylabel("Score")
354
355            fig_name = os.path.join(self.path, f"thresholds_{metric}.pdf")
356            pl.savefig(fig_name)
357            pl.close()
358
359        # Comparing all values at the last epoch
360        pl.figure()
361
362        pl.plot(cst.THRESHOLDS, computed["iou"].T[-1], color = "crimson", label = "IOU")
363        pl.plot(cst.THRESHOLDS, computed["f1"].T[-1], color = "teal", label = "F1 score")
364        pl.plot(cst.THRESHOLDS, computed["ap"].T[-1], color = "purple", label = "Average Precision")
365        pl.plot(cst.THRESHOLDS, computed["pre"].T[-1], color = "goldenrod", label = "Precision")
366        pl.plot(cst.THRESHOLDS, computed["rec"].T[-1], color = "darkorchid", label = "Recall")
367
368
369        # Creating a refined grid and fitting spline
370        large_t = torch.linspace(0.05, 0.95, 1000)
371
372        spline = interp.splrep(
373            x = cst.THRESHOLDS,         # X data is time
374            y = computed["iou"].T[-1],  # Y data is flux
375            k = 4                       # Order of the spline
376        )
377
378        fit_par = interp.splev(large_t, spline)
379        max_fit = large_t[fit_par.argmax()]
380
381        pl.plot(
382            large_t, fit_par,
383            color = "black", alpha = 0.5, zorder = 0,
384            label = f"IOU spline fit (max: {max_fit:.5})"
385        )
386
387        pl.axvline(max_fit.item(), color = "black", alpha = 0.8, dashes = [1, 3, 5, 3])
388
389        pl.xlabel("Threshold value")
390        pl.ylabel("Corresponding score")
391        pl.legend(loc = "best")
392
393        fig_name = os.path.join(self.path, "thresholds_last.pdf")
394        pl.savefig(fig_name)
395        pl.close()

439def analyze(args, cfg: dict):
440    
441    out_name = os.path.join(cfg["path_output"], "2*")
442    outputs = sorted(glob.iglob(out_name))
443    target = []
444
445    # We check the possible values
446    if args == "last":              # If last, we only analyze the last one
447        target.append( outputs[-1] )
448    
449    elif args == "all":             # If all we analyze all the results
450        target = outputs
451
452    else:                           # If not we get the requested one
453        target.append( args )
454    
455    for p in target:
456        result = Output(path = p)
457        result.process()
