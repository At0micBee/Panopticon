 39    """
 40    # Setting up the running environment for training
 41
 42    ## Description
 43
 44    ## Inputs
 45    - `cfg`: The configuration file loaded
 46    - `time_start`: The stamp of the program start
 47
 48    Returns the parameters required to run a training or testing job.
 49    """

 51    def __init__(self, cfg: dict, time_start: datetime):
 52
 53        ##################################################
 54        # Initializing base info
 55        ##################################################
 56
 57        self.logger = logging.getLogger("Panopticon")                           # Fetching the logger for the program
 58        self.logger.info("Entering setup!")
 59
 60        self.cfg = cfg                                                          # Saving the config
 61        self.time_start = time_start                                            # Saving the start time
 62        self.path_out = os.path.join(cfg["path_output"], str(self.time_start))  # Creating a unique name for the run
 63        self.path_models = os.path.join(self.path_out, "model")                 # Creating the path to the models save
 64        self.path_stats = os.path.join(self.path_out, "stats")                  # Creating the path to the stats save
 65        
 66        os.mkdir(self.path_models)                                              # Creating the models output dir
 67        os.mkdir(self.path_stats)                                               # Creating the stats output dir
 68    
 69        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")   # We check the device availability
 70        self.logger.info(f"Found device: {self.device}")
 71
 72        ##################################################
 73        # Preparing the data sets
 74        ##################################################
 75
 76        self.data = LightCurveSet(cfg["path_prepared"])                         # We pass the input files to the Dataset
 77        self.logger.info(f"[Data] Created complete Dataset ({len(self.data)})") # Logging the total length
 78
 79        fracs = cfg["fraction"]                                                 # Getting the fractions from the config
 80        self.logger.info(f"[Data] Requested fractions: {fracs}")                # Logging the values
 81        self.train_set, self.valid_set = random_split(                          # We split into subsets
 82            dataset = self.data,                                                # The sets to break down
 83            lengths = [fracs["train"], fracs["valid"]],                         # The fraction given to each Dataset
 84            generator = torch.Generator().manual_seed(cfg["seed"])              # Manual seed, for reproducibility
 85        )
 86
 87        # If requested we export the dataset files
 88        if cfg["export_sets"]:
 89
 90            # We create a dict and save the names
 91            sets = {
 92                "training": [ os.path.basename(n) for n, l, r in self.train_set ],
 93                "validation": [ os.path.basename(n) for n, l, r in self.valid_set ]
 94            }
 95
 96            # Writing the dict to a pickle file
 97            sets_name = os.path.join(self.path_out, "sets.pt")
 98            with open(sets_name, "wb") as writer:
 99                torch.save(sets, writer)
100
101        self.train_len, self.valid_len = len(self.train_set), len(self.valid_set)
102        self.logger.info(f"[Data] Created training ({self.train_len}) and validation ({self.valid_len}) Subsets")
103
104        ##################################################
105        # Preparing the model
106        ##################################################
107         
108        self.net = load_from_params(name = cfg["model"]).to(self.device)        # Creating the model
109        self.logger.info(f"[Model] Model initialized: {cfg['model']}")
110
111        model_internals = self.net.export_values()                              # Exporting parameters
112
113        with open(os.path.join(self.path_out, "model_internals.toml"), "w+") as mi:
114            toml.dump(model_internals, mi)
115        self.logger.info("[Model] Exported internal values")
116
117        if cfg["base_model"]:                                                   # Checking if we load an existing model
118            with open(cfg["base_model"], "rb") as m:                            # Opening file
119                self.net.load_state_dict(torch.load(m))                         # Assigning weights
120            self.logger.info(f"[Model] Weights loaded from {cfg['base_model']}")
121        
122        else:
123            self.logger.info("[Model] No weights specified")
124        
125        ##################################################
126        # Preparing the data loaders and loss function
127        ##################################################
128
129        self.train_loader = DataLoader(                                         # Training loader
130            self.train_set, **self.cfg["dataloader"]["train"]                   # Passing training set and kwargs
131        )
132        self.logger.info("[Data] Training dataloader created")
133
134        self.valid_loader = DataLoader(                                         # Validation loader
135            self.valid_set, **self.cfg["dataloader"]["valid"]                   # Passing validation set and kwargs
136        )
137        self.logger.info("[Data] Validation dataloader created")
138
139        self.loss_function = Loss(self.device)                                  # Loss function
140        self.logger.info("[Loss] Loss function initialized")

145    def train(self):
146        """
147        # Training the model
148
149        Using the requested dataset (split into a training and validation `Subsets`), we update the model's weights.
150        All the relevant parameters are supplied on the `Run` class, and the `train` function therefore requires
151        no additional parameters.
152        """
153
154        ##################################################
155        # Optim and scheduler
156        ##################################################
157
158        self.optimizer = optimizer[self.cfg["optimizer"]["name"]](              # Optimizer
159            self.net.parameters(), **self.cfg["optimizer"]["kwargs"]            # Passing model and kwargs
160        )
161        self.logger.info(f"[Optimizer] Optimizer initialized ({self.cfg['optimizer']['name']})")
162
163        self.scheduler = scheduler[self.cfg["scheduler"]["name"]](              # Scheduler
164            self.optimizer, **self.cfg["scheduler"]["kwargs"]                   # Passing optimizer and kwargs
165        )
166        self.logger.info(f"[Scheduler] Scheduler initialized ({self.cfg['scheduler']['name']})")
167
168        out_name = os.path.join(self.path_models, f"{0:04}.pt")                 # Creating the proper path
169        with open(out_name, "wb") as writer:                                    # Opening as binary
170            torch.save(self.net.state_dict(), writer)                           # Saving the original model
171        
172        ##################################################
173        # Running the epochs
174        ##################################################
175
176        for epoch in range(1, self.cfg["training_epochs"] + 1):
177
178            time_epoch = datetime.now()
179            time_since_start = time_epoch - self.time_start
180            self.epoch_prog = f"Epoch {epoch:03} / {self.cfg['training_epochs']:03}"
181            self.logger.info(f"{self.epoch_prog} :: starting, {time_since_start} elapsed since start")
182
183            running_loss = 0                                                    # Initial total loss
184            self.net.train()                                                    # Activating training for model
185
186            # We iterate through the training set
187            for _, _, lc, ref in self.train_loader:
188                
189                self.optimizer.zero_grad(set_to_none = True)                    # Zeroing the optimizer
190
191                # Sending LC and ref to the correct type
192                lc = lc.to(device = self.device, non_blocking = True)
193                ref = ref.to(device = self.device, non_blocking = True)
194
195                prediction = self.net(lc)                                       # Computing the prediction from the model
196
197                loss = self.loss_function(prediction, ref)                      # Estimating the loss compared to reality
198
199                loss.backward()                                                 # Propagating the loss
200                self.optimizer.step()                                           # Incrementing the model
201
202                running_loss += loss.item()                                     # Computing full loss of epoch
203            
204            self.logger.info(f"{self.epoch_prog} :: finished training, entering evaluation (average loss: {running_loss})")
205
206            out_name = os.path.join(self.path_models, f"{epoch:04}.pt")         # Creating the proper path
207            with open(out_name, "wb") as writer:                                # Opening as binary
208                torch.save(self.net.state_dict(), writer)                       # Saving the original model
209            self.logger.info(f"{self.epoch_prog} :: Exported model parameters")
210
211            self.validate(epoch)                                                # We call the validation routine
212
213            if self.cfg["scheduler"]["use"]:                                    # Checking if we need the scheduler
214
215                if self.cfg["scheduler"]["name"] == "ReduceLROnPlateau":
216                    self.scheduler.step(loss)                                   # Incrementing the scheduler
217                else:
218                    self.scheduler.step()
219                
220                current_lr = self.scheduler.get_lr()
221                self.logger.info(f"[Scheduler] {self.epoch_prog} :: Learning rate adjusted -> {current_lr}")
222            
223            self.logger.info(f"{self.epoch_prog} :: End of epoch")
224
225

229    @torch.no_grad()                                                            # Disabling gradient computation
230    def validate(self, epoch: int):
231        """
232        # Testing a model on a given `Subset`
233
234        We go through the validation dataset to test the performance of the model.
235        In standard operation, the function is called at every training epoch.
236
237        The function uses the `@torch.no_grad()` decorator to disable `Tensor` gradient computation,
238        which isn't necessary during testing. The model is also flipped in `model.eval()` mode at the
239        beginning of the function, and set back to `model.train()` mode before returning.
240
241        - `epoch`: the current epoch counter, for book-keeping.
242        """
243
244        self.net.eval()                                                         # Setting mode to evaluation mode
245        self.logger.info(f"{self.epoch_prog} :: Entered validation step")
246
247        stats = {}                                                              # Creating the dict to keep track of data
248        running_loss = 0                                                        # Initial total loss
249
250        for f_name, sims_params, lc, ref in self.valid_loader:                  # Iterating through the loader
251
252            stats[tuple(f_name)] = {}                                           # Creating the dict at this value
253
254            lc = lc.to(device = self.device, non_blocking = True)               # Sending lc tensor to the right device
255            ref = ref.to(device = self.device, non_blocking = True)             # Same for the reference
256            ref_int = ref.to(dtype = int, non_blocking = True)                  # Creating the reference version with int
257
258            # Prediction and associated loss
259            prediction = self.net(lc)                                           # Model prediction
260            loss = self.loss_function(prediction, ref).detach().cpu()           # Estimating the loss
261            running_loss += loss.item()                                         # Computing full loss of epoch
262
263            # We compute the metrics for each threshold
264            for thr in cst.THRESHOLDS:
265
266                thr = thr.item()                                                # Extracting the float in the tensor
267
268                # Computing metrics
269                precision = binary_precision(prediction, ref_int, thr, "samplewise").detach().cpu()     # Computing precision
270                recall = binary_recall(prediction, ref_int, thr, "samplewise").detach().cpu()           # Computing the recall
271                f1 = binary_f1_score(prediction, ref_int, thr, "samplewise").detach().cpu()             # Computing the F1
272                ap = binary_average_precision(prediction, ref_int, [thr]).detach().cpu()                # Computing the average precision
273                cm = binary_confusion_matrix(prediction, ref_int, thr).detach().cpu()                   # Calculating the confusion matrix
274                iou = binary_jaccard_index(prediction, ref_int, thr).detach().cpu()                     # Computing the IOU
275                radii, sample = transit_recovery(prediction, ref, sims_params, thr)                     # List of the retrieved radii
276
277                stats[tuple(f_name)][thr] = {                                   # We create a dict with the params
278                    "loss": loss,                                               # Adding the loss to the dict
279                    "precision": precision,                                     # Precision
280                    "recall": recall,                                           # Recall
281                    "f1": f1,                                                   # F1
282                    "ap": ap,                                                   # Average precision
283                    "cm": cm,                                                   # Confusion matrix
284                    "iou": iou,                                                 # IOU
285                    "radii": radii,                                             # The recovered radii
286                    "sample": sample                                            # The radii of the sample
287                }
288
289        self.logger.info(f"{self.epoch_prog} :: Validation finished, average loss: {running_loss / len(self.valid_loader)}")
290
291        out_name = os.path.join(self.path_stats, f"{epoch:04}.pt")              # Creating the proper path
292        with open(out_name, "wb") as writer:                                    # Opening as Binary
293            torch.save(stats, writer)                                           # Saving the parameters
294
295        self.net.train()                                                        # Returning the model to training
